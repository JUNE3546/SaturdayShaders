uniform vec2 u_resolution;
uniform float u_time;
float left;
float right;
float top;
float bottom;
float speed;

#ifdef GL_ES
precision mediump float;
#endif


void main() {

    vec2 st = gl_FragCoord.xy/u_resolution;

    speed = 0.2;
    left = abs(sin(u_time * speed )-0.1);
    right = abs(sin(u_time * speed)+0.1);
    top = 0.35;
    bottom = 0.65;

    float y;
    if(right > left){
        y = step(st.x,right) * step(left, st.x);
    } else if(right < left){
        y = step(st.x,left) * step(right, st.x);    
    }
    float x;
    if(top>bottom){
        x = step(st.y, top) * step(bottom, st.y);
    } else if(top < bottom){
        x = step(st.y, bottom) * step(top, st.y);
        
    }  

    vec3 color = vec3(y * x);

    gl_FragColor = vec4(color,1.0);
}
